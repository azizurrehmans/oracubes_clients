create table student_course (
			student_id 	number(6),
			course_id 	varchar2(50),
			fee 		number(6),
			start_date 	date,
			end_date 	date,
			diploma 	varchar2(50),
			marks 		number(6),
			timing 		varchar2(50),
			fdatetime 	date default sysdate,
			user_id 	varchar2(50),
			status_id 	varchar2(50) default 'HOLD',
			constraint student_course_course_fk 	foreign key (course_id) references course(course_id),
			constraint student_course_user_fk 		foreign key (user_id) references users(user_id),
			constraint student_course_student_fk 	foreign key (student_id) references student(student_id));
